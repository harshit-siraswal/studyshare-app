1. System Architecture
1.1 High-Level Architecture

Presentation Layer → Domain Layer → Data Layer → External Services

Domain Layer: Use Cases, Business Logic, Repository Interfaces
Data Layer: Repository Implementations, Local DB, Remote API clients

3.3 State Management
Each feature uses a ViewModel to manage UI state. States are represented as sealed classes or data classes and exposed as StateFlows for Compose to observe.
Example State Pattern:
sealed class ResourceState {
    object Loading : ResourceState()
    data class Success(val resources: List<Resource>) : ResourceState()
    data class Error(val message: String) : ResourceState()
}

class ResourceViewModel @Inject constructor(
    private val getResourcesUseCase: GetResourcesUseCase
) : ViewModel() {
    
    private val _state = MutableStateFlow<ResourceState>(ResourceState.Loading)
    val state: StateFlow<ResourceState> = _state.asStateFlow()
    
    fun loadResources(filters: ResourceFilters) {
        viewModelScope.launch {
            getResourcesUseCase(filters)
                .catch { _state.value = ResourceState.Error(it.message ?: "Unknown error") }
                .collect { _state.value = ResourceState.Success(it) }
        }
    }
}

4.2 Repository Pattern
Repositories act as single source of truth, abstracting data sources from the domain layer. They implement cache-first strategies where appropriate.
interface ResourceRepository {
    fun getResources(filters: ResourceFilters): Flow<Result<List<Resource>>>
    suspend fun getResourceById(id: String): Result<Resource>
    suspend fun uploadResource(resource: CreateResourceRequest): Result<Resource>
    suspend fun voteResource(id: String, vote: Vote): Result<Unit>
    suspend fun bookmarkResource(id: String): Result<Unit>
}

class ResourceRepositoryImpl @Inject constructor(
    private val remoteDataSource: ResourceRemoteDataSource,
    private val localDataSource: ResourceLocalDataSource,
    private val networkMonitor: NetworkMonitor
) : ResourceRepository {
    
    override fun getResources(filters: ResourceFilters): Flow<Result<List<Resource>>> = flow {
        // Cache-first strategy
        val cachedData = localDataSource.getResources(filters)
        if (cachedData.isNotEmpty()) {
            emit(Result.success(cachedData))
        }
        
        if (networkMonitor.isOnline()) {
            val remoteData = remoteDataSource.getResources(filters)
            localDataSource.saveResources(remoteData)
            emit(Result.success(remoteData))
        }
    }
}

5.3 Error Handling
sealed class ApiError {
    data class NetworkError(val exception: IOException) : ApiError()
    data class HttpError(val code: Int, val message: String) : ApiError()
    data class UnknownError(val throwable: Throwable) : ApiError()
}

suspend fun <T> safeApiCall(apiCall: suspend () -> T): Result<T> {
    return try {
        Result.success(apiCall())
    } catch (e: IOException) {
        Result.failure(ApiError.NetworkError(e))
    } catch (e: HttpException) {
        Result.failure(ApiError.HttpError(e.code(), e.message()))
    } catch (e: Exception) {
        Result.failure(ApiError.UnknownError(e))
    }
}

class ChatRealtimeService @Inject constructor(

) {
    fun subscribeToRoom(roomId: String): Flow<ChatMessage> = callbackFlow {
        val channel = supabaseClient.realtime.channel("room_$roomId")
        
        channel.on(RealtimeChannel.INSERT) { message ->

6.1 Firebase Authentication Flow
1. User taps "Sign in with Google"
2. Google Sign-In SDK presents account picker
3. User selects account and authorizes
4. Firebase SDK receives ID token
5. App sends token to backend /auth/verify
6. Backend validates with Firebase Admin SDK
7. Backend checks email domain for verification
8. Backend returns session token
9. App stores token in Android Keystore
10. Subsequent requests include token in Authorization header
6.2 Token Management
class TokenManager @Inject constructor(
    private val keyStore: KeyStore,
    private val encryptedPrefs: EncryptedSharedPreferences
) {
    suspend fun saveToken(token: String) {
        val encryptedToken = encryptToken(token)
        encryptedPrefs.edit {
            putString(KEY_AUTH_TOKEN, encryptedToken)
        }
    }
    
    suspend fun getToken(): String? {
        val encryptedToken = encryptedPrefs.getString(KEY_AUTH_TOKEN, null)
        return encryptedToken?.let { decryptToken(it) }
    }
    
    suspend fun refreshToken() {
        val currentUser = Firebase.auth.currentUser
        currentUser?.getIdToken(true)?.await()?.token?.let { newToken ->
            saveToken(newToken)
        }
    }
    
    private fun encryptToken(token: String): String {
        // Use Android Keystore for encryption
        val cipher = Cipher.getInstance(TRANSFORMATION)
        cipher.init(Cipher.ENCRYPT_MODE, getSecretKey())
        return Base64.encodeToString(cipher.doFinal(token.toByteArray()), Base64.DEFAULT)
    }
}

class BiometricAuthManager @Inject constructor(
    private val context: Context
) {
    fun isBiometricAvailable(): Boolean {
        val biometricManager = BiometricManager.from(context)
        return biometricManager.canAuthenticate(BIOMETRIC_STRONG) == BiometricManager.BIOMETRIC_SUCCESS
    }
    
    suspend fun authenticateWithBiometric(): Result<Unit> = suspendCoroutine { continuation ->
        val executor = ContextCompat.getMainExecutor(context)
        val biometricPrompt = BiometricPrompt(
            context as FragmentActivity,
            executor,
            object : BiometricPrompt.AuthenticationCallback() {
                override fun onAuthenticationSucceeded(result: BiometricPrompt.AuthenticationResult) {
                    continuation.resume(Result.success(Unit))
                }
                
                override fun onAuthenticationError(errorCode: Int, errString: CharSequence) {
                    continuation.resume(Result.failure(Exception(errString.toString())))
                }
            }
        )
        
        val promptInfo = BiometricPrompt.PromptInfo.Builder()
            .setTitle("Unlock MyStudySpace")
            .setSubtitle("Use your fingerprint or face to sign in")
            .setNegativeButtonText("Use password")
            .build()
        
        biometricPrompt.authenticate(promptInfo)
    }
}

class TimerService : Service() {
    private var timeRemaining = 0L
    private var isRunning = false
    private val scope = CoroutineScope(SupervisorJob() + Dispatchers.Main)
    
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        when (intent?.action) {
            ACTION_START -> startTimer()
            ACTION_PAUSE -> pauseTimer()
            ACTION_RESET -> resetTimer()
        }
        return START_STICKY
    }
    
    private fun startTimer() {
        isRunning = true
        showNotification()
        scope.launch {
            while (isRunning && timeRemaining > 0) {
                delay(1000)
                timeRemaining -= 1000
                updateNotification(timeRemaining)
            }
            if (timeRemaining == 0) {
                playAlarm()
                showCompletionNotification()
            }
        }
    }
    
    private fun showNotification() {
        val notification = NotificationCompat.Builder(this, TIMER_CHANNEL_ID)
            .setContentTitle("Study Timer")
            .setContentText(formatTime(timeRemaining))
            .setSmallIcon(R.drawable.ic_timer)
            .setPriority(NotificationCompat.PRIORITY_HIGH)
            .setOngoing(true)
            .addAction(R.drawable.ic_pause, "Pause", pausePendingIntent)
            .addAction(R.drawable.ic_stop, "Stop", stopPendingIntent)
            .build()
        
        startForeground(TIMER_NOTIFICATION_ID, notification)
    }
}

8.4 Background Task Optimization
@HiltWorker
class DownloadWorker @AssistedInject constructor(
    @Assisted appContext: Context,
    @Assisted workerParams: WorkerParameters,
    private val downloadRepository: DownloadRepository
) : CoroutineWorker(appContext, workerParams) {
    
    override suspend fun doWork(): Result {
        val resourceId = inputData.getString(KEY_RESOURCE_ID) ?: return Result.failure()
        
        setForeground(createForegroundInfo())
        
        return try {
            downloadRepository.downloadResource(resourceId) { progress ->
                setProgress(workDataOf(KEY_PROGRESS to progress))
            }
            Result.success()
        } catch (e: Exception) {
            if (runAttemptCount < 3) {
                Result.retry()
            } else {
                Result.failure()
            }
        }
    }
    
    private fun createForegroundInfo(): ForegroundInfo {
        val notification = NotificationCompat.Builder(applicationContext, DOWNLOAD_CHANNEL_ID)
            .setContentTitle("Downloading resource")
            .setProgress(100, 0, false)
            .setSmallIcon(R.drawable.ic_download)
            .build()
        
        return ForegroundInfo(DOWNLOAD_NOTIFICATION_ID, notification)
    }
}

// Schedule download

            .setRequiredNetworkType(NetworkType.UNMETERED) // Wi-Fi only
            .setRequiresStorageNotLow(true)
            .build()
    )
    .build()

@ExperimentalCoroutinesTest
class ResourceViewModelTest {
    
    @get:Rule
    val instantExecutorRule = InstantTaskExecutorRule()
    
    private val testDispatcher = StandardTestDispatcher()
    private val mockGetResourcesUseCase: GetResourcesUseCase = mockk()
    private lateinit var viewModel: ResourceViewModel
    
    @Before
    fun setup() {
        Dispatchers.setMain(testDispatcher)
        viewModel = ResourceViewModel(mockGetResourcesUseCase)
    }
    
    @After
    fun tearDown() {
        Dispatchers.resetMain()
    }
    
    @Test
    fun `loadResources should update state to Success when use case returns data`() = runTest {
        // Given
        val resources = listOf(Resource(id = "1", title = "Test"))
        coEvery { mockGetResourcesUseCase(any()) } returns flowOf(resources)
        
        // When
        viewModel.loadResources(ResourceFilters.default())
        testDispatcher.scheduler.advanceUntilIdle()
        
        // Then
        val state = viewModel.state.value
        assertThat(state).isInstanceOf(ResourceState.Success::class.java)
        assertThat((state as ResourceState.Success).resources).isEqualTo(resources)
    }
    
    @Test
    fun `loadResources should update state to Error when use case throws exception`() = runTest {
        // Given
        val errorMessage = "Network error"
        coEvery { mockGetResourcesUseCase(any()) } throws IOException(errorMessage)
        
        // When
        viewModel.loadResources(ResourceFilters.default())
        testDispatcher.scheduler.advanceUntilIdle()
        
        // Then
        val state = viewModel.state.value
        assertThat(state).isInstanceOf(ResourceState.Error::class.java)
        assertThat((state as ResourceState.Error).message).isEqualTo(errorMessage)
    }
}
